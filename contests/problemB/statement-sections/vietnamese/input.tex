\begin{itemize}
\item Dòng 1: hai số nguyên $N, M$ ($1 \le N, M \le 161\,616$).
\item $N$ dòng tiếp: \texttt{start\_time[T]} và \texttt{commit\_time[T]} của từng giao dịch $T$ ($1 \le T \le N$).
\item $M$ dòng tiếp: \texttt{create\_tx[j]} và \texttt{delete\_tx[j]} (ghi $-1$ nếu chưa bị xóa) cho từng tuple $j$ ($1 \le j \le M$).
\end{itemize}
